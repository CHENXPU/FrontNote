  "(^\\d{15}$)|^(\\d{6})(\\d{4})(\\d{2})(\\d{2})(\\d{3})([0-9]|X|x)$",
); // 十五位或十八位身份证
/(^\d{15}$)|^(\d{6})(\d{4})(\d{2})(\d{2})(\d{3})([0-9]|X|x)$/.test(
  "444444199501011111",
); // true
